        displayTodoList()
        function displayTodoList()
        {
            let todoListString=""
            todoListArray.forEach(
                (todoItem)=>{
                      todoListString= todoListString + `<p> ${todoItem.todoTask} </p> <p> ${todoItem.todoTaskDate}</p> <button class='delete-button'> Delete </button> `
                }
            )
            showList.innerHTML=todoListString

             let deleteButtons=document.querySelectorAll('.delete-button')
            console.log(deleteButtons)

        deleteButtons.forEach(
            (singleDeleteButton,index)=>{
                        singleDeleteButton.addEventListener('click',()=>{
                            todoListArray.splice(index,1)
                            console.log(todoListArray)
                            displayTodoList()
                        })
            }
        )
        }


        function addToList()
        {
           
            let inputTextElement=document.querySelector('#input-text') 
            let inputDateElement=document.querySelector('#input-date') 
            let todoTask=inputTextElement.value;     //task3               
            let todoTaskDate=inputDateElement.value; //29-sept-2025

            console.log(todoTask)
            console.log(todoTaskDate)
            let todoObject={todoTask,todoTaskDate}  //{todoTask:task3 ,todoTaskDate:29-sept-2025}
            
            let booleanResult=checkForDuplicateTask(todoObject)
            if(todoTask=="" || todoTaskDate=="")
            alert('Both fields are compulsory')
            else if(!booleanResult)
            {
            todoListArray.push(todoObject)
            inputTextElement.value=""
            inputTextElement.focus()
            }
            else
            {
                alert ('Item Allready existing')
                return
            }
            console.log( todoListArray)
            displayTodoList()
        }

        function checkForDuplicateTask(todoObject)
        {
                for(let i=0;i<todoListArray.length;i++)
                {
                    if(JSON.stringify(todoListArray[i])===JSON.stringify(todoObject))
                    return true;
                }
                return false;
        }
