Text inside HTML is treated as a Text Node, not a string.
to read that text we use ".nodeValue"

<li id="one">Buy Milk<button>Delete</button></li>															   Type
												← first child (TEXT NODE)					Node.TEXT_NODE (3)
												← second child (ELEMENT NODE)				Node.ELEMENT_NODE (1)
																							Node.attributes   (2)

Read attributes
const liElement=document.getElementById("one")
const attrNode = liElement.getAttributeNode("id");
console.log(attrNode.nodeType)   //2
console.log(attrNode)           //id="one"

console.log(liElement.innerText);  					//Buy Milk Delete
console.log(liElement.firstChild.nodeValue);		//Buy Milk

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Create & Remove</title>
</head>
<body>

    <input id="task" placeholder="Enter task" />
    <button onclick="addTask()">Add Task</button>

    <ul id="tasks"></ul>

    <script>
    function addTask() {
        const taskInput = document.getElementById("task");
        const taskText = taskInput.value.trim();

        //prevent empty input
        if (taskText === "") {
            alert("Please enter a task");
            return;
        }

        const tasksList = document.getElementById("tasks");
        const existingTasks = tasksList.getElementsByTagName("li");

        // check duplicates
        for (let li of existingTasks) {
            if (li.firstChild.nodeValue === taskText) {
                alert("Task already exists!");
                taskInput.value = "";
                return;
            }
        }

        // create li
        const li = document.createElement("li");
        li.innerText = taskText;

        // delete button
        const btn = document.createElement("button");
        btn.innerText = "Delete";

        li.appendChild(btn);
        tasksList.appendChild(li);

        // clear input
        taskInput.value = "";

        // delete logic
        btn.onclick = function () {
            li.remove();
        };
    }
</script>


</body>
</html>
